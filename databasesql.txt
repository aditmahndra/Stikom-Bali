-- Buat database
CREATE DATABASE IF NOT EXISTS itb_stikom;
USE itb_stikom;

-- Tabel users (untuk login dan registrasi)
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nama_lengkap VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    PASSWORD VARCHAR(255) NOT NULL,
    no_ponsel VARCHAR(15),
    ROLE ENUM('mahasiswa', 'admin') DEFAULT 'mahasiswa',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabel pendaftaran S2
CREATE TABLE pendaftaran_s2 (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    nama_lengkap VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    tanggal_lahir DATE NOT NULL,
    jk ENUM('Laki-laki', 'Perempuan') NOT NULL,
    prodi VARCHAR(100) NOT NULL,
    no_hp VARCHAR(15) NOT NULL,
    nim VARCHAR(20) UNIQUE,
    angkatan YEAR,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Tabel pendaftaran S1 dan D3
CREATE TABLE pendaftaran_s1_d3 (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    nama_lengkap VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    tanggal_lahir DATE NOT NULL,
    jk ENUM('Laki-laki', 'Perempuan') NOT NULL,
    prodi VARCHAR(100) NOT NULL,
    no_hp VARCHAR(15) NOT NULL,
    nim VARCHAR(20) UNIQUE,
    angkatan YEAR,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Tabel pendaftaran Dual Degree
CREATE TABLE pendaftaran_dual_degree (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    nama_lengkap VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    tanggal_lahir DATE NOT NULL,
    jk ENUM('Laki-laki', 'Perempuan') NOT NULL,
    prodi VARCHAR(100) NOT NULL,
    no_hp VARCHAR(15) NOT NULL,
    nim VARCHAR(20) UNIQUE,
    angkatan YEAR,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Trigger untuk generate NIM otomatis
DELIMITER //
CREATE TRIGGER generate_nim_s2 
BEFORE INSERT ON pendaftaran_s2
FOR EACH ROW
BEGIN
    IF NEW.nim IS NULL THEN
        SET NEW.nim = CONCAT('S2', LPAD(FLOOR(RAND() * 99999), 5, '0'));
    END IF;
    IF NEW.angkatan IS NULL THEN
        SET NEW.angkatan = YEAR(CURDATE());
    END IF;
END//

CREATE TRIGGER generate_nim_s1_d3 
BEFORE INSERT ON pendaftaran_s1_d3
FOR EACH ROW
BEGIN
    IF NEW.nim IS NULL THEN
        SET NEW.nim = CONCAT('S1', LPAD(FLOOR(RAND() * 99999), 5, '0'));
    END IF;
    IF NEW.angkatan IS NULL THEN
        SET NEW.angkatan = YEAR(CURDATE());
    END IF;
END//

CREATE TRIGGER generate_nim_dual_degree 
BEFORE INSERT ON pendaftaran_dual_degree
FOR EACH ROW
BEGIN
    IF NEW.nim IS NULL THEN
        SET NEW.nim = CONCAT('DD', LPAD(FLOOR(RAND() * 99999), 5, '0'));
    END IF;
    IF NEW.angkatan IS NULL THEN
        SET NEW.angkatan = YEAR(CURDATE());
    END IF;
END//
DELIMITER ;
